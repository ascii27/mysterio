# M3.3 — TTS Voice Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Synthesize each generated narrative into an MP3 via OpenAI TTS, store it on local disk, and play it back in the web app via a native `<audio controls>` element.

**Architecture:** New `services/tts/` module under the server with a swappable `TTSProvider` interface + OpenAI adapter + filesystem storage helper. A dedicated `ttsAgent` orchestrates synthesize → store with one retry. The generation orchestrator transitions `synthesizing` between narrative success and `ready`, soft-failing to `ready` with `audio_path: null` if TTS exhausts retries. Frontend renders one conditional native `<audio>` element on the PlaybackScreen.

**Tech Stack:** TypeScript, Vitest, OpenAI Node SDK, fastify-static (already wired), React (web).

**Reference:** `docs/superpowers/specs/2026-05-23-m3.3-tts-voice-design.md`

**Branch:** `feat/m3.3-tts-voice` (already created, design doc committed at `61fa8bf`).

**Pre-decided by env scaffolding (do not relitigate):**
- Provider: OpenAI (`OPENAI_API_KEY` already required at boot)
- Model: `tts-1-hd` (`OPENAI_TTS_MODEL` env default)
- Voice: `fable` (`OPENAI_TTS_VOICE` env default, single narrator)
- Storage: local FS at `./data/audio` (`AUDIO_DIR` env default)
- Static serve: `routes/static.ts` already serves `/audio/*` with `cacheControl: true, maxAge: "365d", immutable: true`
- Schema: `mysteries.audio_path` column + `'synthesizing'` status both exist
- DTO: `GET /api/mysteries/:id` already returns `audio_url: row.audio_path ? '/audio/${path}' : null`

Create `apps/server/src/services/tts/openai.ts`:
```ts
import OpenAI from "openai";
import { loadEnv } from "../../config/env.js";
import type { SynthesizeOpts, TTSProvider } from "./provider.js";

let _client: OpenAI | null = null;
function getOpenAi(): OpenAI {
  if (_client) return _client;
  const env = loadEnv();
  _client = new OpenAI({ apiKey: env.OPENAI_API_KEY, timeout: 30_000 });
  return _client;
}

export class OpenAiProvider implements TTSProvider {
  async synthesize(text: string, opts: SynthesizeOpts): Promise<Buffer> {
    const client = getOpenAi();
    const res = await client.audio.speech.create({
      model: opts.model,
      voice: opts.voice,
      input: text,
      response_format: "mp3",
    });
    const ab = await res.arrayBuffer();
    return Buffer.from(ab);
  }
}
```

> **Note:** The `voice` field on OpenAI's TS types is a union of literal strings. If TS complains about `opts.voice` being too wide, cast at the call site: `voice: opts.voice as "fable"`. The runtime accepts any of OpenAI's documented voices regardless of the static type.

import { mkdir, writeFile } from "node:fs/promises";
import { resolve } from "node:path";
import { loadEnv } from "../../config/env.js";

export async function writeAudio(mysteryId: string, buf: Buffer): Promise<string> {
  const env = loadEnv();
  const audioDir = resolve(env.AUDIO_DIR);
  await mkdir(audioDir, { recursive: true });
  const fileName = `${mysteryId}.mp3`;
  await writeFile(`${audioDir}/${fileName}`, buf);
  return fileName;
}
```

import { loadEnv } from "../../../config/env.js";
import { logger } from "../../../utils/logger.js";
import { defaultProvider } from "../../tts/index.js";
import { writeAudio } from "../../tts/storage.js";

export interface TtsAgentInput {
  mysteryId: string;
  narrativeText: string;
}

export type TtsAgentResult =
  | { ok: true; audioPath: string }
  | { ok: false; error: string };

const MAX_ATTEMPTS = 2;

export async function runTtsAgent(input: TtsAgentInput): Promise<TtsAgentResult> {
  const env = loadEnv();
  let lastErr = "";
  let buf: Buffer | null = null;

  for (let attempt = 1; attempt <= MAX_ATTEMPTS; attempt++) {
    try {
      buf = await defaultProvider.synthesize(input.narrativeText, {
        model: env.OPENAI_TTS_MODEL,
        voice: env.OPENAI_TTS_VOICE,
      });
      break;
    } catch (e) {
      lastErr = e instanceof Error ? e.message : String(e);
      logger.info("tts_provider_failed", { mystery_id: input.mysteryId, attempt, error: lastErr });
    }
  }

  if (!buf) {
    return { ok: false, error: lastErr };
  }

  try {
    const audioPath = await writeAudio(input.mysteryId, buf);
    return { ok: true, audioPath };
  } catch (e) {
    const err = e instanceof Error ? e.message : String(e);
    logger.info("tts_storage_failed", { mystery_id: input.mysteryId, error: err });
    return { ok: false, error: err };
  }
}
```
